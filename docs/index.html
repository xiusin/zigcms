概述

ZigCMS SQL ORM 是一个为 Zig 语言设计的现代化 ORM 框架，灵感来自 Laravel Eloquent 和 GORM。 它提供了类型安全、高性能的数据库抽象层，支持多种数据库系统。

✨ 核心特性
  • 类型安全：利用 Zig 的编译时特性，在编译阶段捕获错误
  • 零反射：使用 comptime 元编程，无运行时反射开销
  • 多数据库支持：统一 API，轻松切换 MySQL、SQLite、PostgreSQL
  • 线程安全：内置 Mutex 保护，支持并发访问
  • 内存安全：明确的内存管理策略，避免内存泄漏

安装

前置条件

  • Zig 0.15.1 或更高版本
  • 数据库驱动（根据使用的数据库）：
    • MySQL: libmysqlclient
    • SQLite: libsqlite3
    • PostgreSQL: pg 库

# MySQL

将 ORM 模块添加到您的项目中：

// 在 build.zig 中
const sql_module = b.addModule("sql", .{

});

exe.root_module.addImport("sql", sql_module);

快速开始

1. 导入模块

const std = @import("std");
const sql = @import("sql");

pub fn main() !void {
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    const allocator = gpa.allocator();

    // 创建数据库连接
    var db = try sql.Database.sqlite(allocator, ":memory:");
    defer db.deinit();

    std.debug.print("数据库连接成功!\n", .{});
}

2. 定义模型

const User = sql.Model(struct {
    pub const table = "users";

    name: []const u8,
    email: []const u8,
    age: u32,
    active: bool,

声明模型

模型是数据库表的 Zig 结构体表示。使用 sql.Model() 函数定义模型：

基础模型

const User = sql.Model(struct {
    pub const table = "users";          // 表名
    pub const primary_key = "id";       // 主键字段
    
    id: u64,
    name: []const u8,
    email: []const u8,
    created_at: ?[]const u8,
});

支持的字段类型

Zig 类型 SQL 类型 说明
u64, u32, i64, i32 INTEGER, BIGINT 整数类型
f64, f32 FLOAT, DOUBLE 浮点数类型
bool BOOLEAN, TINYINT 布尔类型
[]const u8 TEXT, VARCHAR 字符串类型
?T 允许 NULL 可选类型

可选字段

const Post = sql.Model(struct {
    pub const table = "posts";
    
    id: u64,
    user_id: u64,
    title: []const u8,
    content: ?[]const u8,        // 可以为 NULL
    published_at: ?[]const u8,   // 可以为 NULL
    views: u32 = 0,              // 默认值
});

关联模型（新 API）

支持 Laravel Eloquent 风格的关联查询：

一对多关联 (HasMany)

// 定义模型
const User = orm.define(struct {
    pub const table_name = "users";
    id: u64,
    name: []const u8,
});

const Post = orm.define(struct {
    pub const table_name = "posts";
    id: u64,
    user_id: u64,
    title: []const u8,
});

// 设置默认连接
User.use(&db);
Post.use(&db);

// 获取用户的所有文章 (HasMany)
if (try User.Find(1)) |*u| {
    var user = u.*;
    defer User.freeModel(allocator, &user);
    
    // 使用 HasMany 获取关联
    var rel = User.HasMany(Post.Model, "user_id", user.id);
    const posts = try rel.get();
    defer Post.freeModels(allocator, posts);
    
    std.debug.print("用户 {s} 有 {d} 篇文章\n", .{user.name, posts.len});
}

属于关联 (BelongsTo)

// 获取文章的作者 (BelongsTo)
if (try Post.Find(1)) |*p| {
    var post = p.*;
    defer Post.freeModel(allocator, &post);
    
    // 使用 BelongsTo 获取关联
    var rel = Post.BelongsTo(User.Model, post.user_id);
    if (try rel.first()) |*u| {
        var author = u.*;
        defer User.freeModel(allocator, &author);
        std.debug.print("文章作者: {s}\n", .{author.name});
    }
}

关联查询方法

// 带条件的关联查询
var rel = User.HasMany(Post.Model, "user_id", user_id);
_ = rel.where("status", "=", "published")
       .orderBy("created_at", .desc)
       .limit(10);
const posts = try rel.get();

// 获取关联数量
var rel2 = User.HasMany(Post.Model, "user_id", user_id);
const count = try rel2.count();
rel2.deinit();

// 检查是否存在关联
var rel3 = User.HasMany(Post.Model, "user_id", user_id);
const has_posts = try rel3.exists();  // 自动 deinit

// 获取关联并返回 List
var rel4 = User.HasMany(Post.Model, "user_id", user_id);
var list = try rel4.collect();
defer list.deinit();

使用 Model.Model 获取原始结构体类型用于关联查询。

查询构建器

查询构建器提供了流畅的链式 API 来构建复杂查询：

完美 Laravel 风格（无 db 参数，推荐）

// 初始化时设置默认数据库连接（只需一次）
User.use(&db);
Product.use(&db);

// 现在可以像 Laravel 一样调用！完全不需要 db 参数！
var q = User.Where("age", ">", 18);
defer q.deinit();

const users = try q.orderBy("name", .asc).limit(10).get();
defer User.freeModels(allocator, users);

// 完美的 Laravel 风格方法
var q1 = User.WhereEq("status", "active");     // WHERE status = 'active'
var q2 = User.WhereLike("name", "%张%");       // WHERE name LIKE '%张%'
var q3 = User.WhereIn("id", &.{1, 2, 3});      // WHERE id IN (1, 2, 3)
var q4 = User.WhereNull("deleted_at");         // WHERE deleted_at IS NULL
var q5 = User.WhereNotNull("email");           // WHERE email IS NOT NULL
var q6 = User.Latest();                        // ORDER BY created_at DESC
var q7 = User.Oldest();                        // ORDER BY created_at ASC
var q8 = User.Take(10);                        // LIMIT 10
var q9 = User.Paginate(1, 20);                 // 第1页，每页20条
var qa = User.OrderBy("name", .asc);           // ORDER BY name ASC

// CRUD 操作也无需 db 参数
const user = try User.Find(1);                 // 查找
const total = try User.Count();                // 统计
var list = try User.Collect();                 // 获取全部
const first = try User.First();                // 第一条
const created = try User.Create(.{...});       // 创建
try User.Update(1, .{...});                    // 更新
try User.Destroy(1);                           // 删除

// 使用 withDB 指定连接（用于事务等场景）
_ = try User.withDB(&tx).Create(.{...});       // 事务中创建
_ = try User.withDB(&tx).Update(1, .{...});    // 事务中更新
_ = try User.withDB(&tx).Destroy(1);           // 事务中删除

// 链式调用
var results = try User.Where("active", "=", 1)
    .whereGte("age", 18)
    .orderBy("created_at", .desc)
    .take(10)
    .collect();
defer results.deinit();

  • 批量操作: 使用事务包装批量插入/更新
  • 预加载: 使用 JOIN 避免 N+1 查询
  • 指定字段: 避免 SELECT *，只查询需要的字段
  • 连接复用: 复用数据库连接，避免频繁创建
  • 查询分析: 使用 EXPLAIN 分析慢查询
  • 监控: 记录和分析查询性能

数据库迁移

ORM 提供编译时 Schema 生成功能，可自动从模型定义生成建表语句，支持 MySQL、SQLite、PostgreSQL 三种方言。

单模型迁移

const orm_models = @import("domain/entities/orm_models.zig");

// 生成建表 SQL（编译期，零运行时开销）
const create_sql = orm_models.Admin.createTableSql(.mysql);
const drop_sql = orm_models.Admin.dropTableSql();

// 执行建表/删表
try orm_models.Admin.createTable(&db);
try orm_models.Admin.dropTable(&db);

批量迁移（Migrator）

const sql = @import("application/services/sql/mod.zig");
const Migrator = sql.Migrator;

// 批量创建表
try Migrator.createAll(&db, .{
    orm_models.Admin,
    orm_models.Article,
    orm_models.Category,
});

// 批量删除表（按依赖倒序）
try Migrator.dropAll(&db, .{
    orm_models.Category,
    orm_models.Article,
    orm_models.Admin,
});

// 刷新表（删除后重建）
try Migrator.refreshAll(&db, 
    .{ Category, Article, Admin },  // drop order
    .{ Admin, Article, Category },  // create order
);

便捷函数

// 迁移所有表
try orm_models.migrate(&db);

// 回滚所有表
try orm_models.rollback(&db);

// 刷新所有表
try orm_models.refresh(&db);

// 打印所有建表 SQL（调试用）
orm_models.printMigrationSql(.mysql);

聚合函数

支持 Laravel 风格的聚合查询：SUM、AVG、MIN、MAX、COUNT。

基本用法

var query = Product.query(&db);
defer query.deinit();

// SUM
const total = try query.whereEq("category", "电子").sum("price");
// AVG
const avg_price = try query.avg("price");
// MIN / MAX
const min_price = try query.min("price");
const max_price = try query.max("price");
// COUNT
const count = try query.count();

OR 条件

var query = User.query(&db);
defer query.deinit();

// WHERE status = 1 OR role = 'admin'
_ = query.whereEq("status", 1).orWhereEq("role", "admin");

分组条件 (Laravel 闭包风格)

类似 Laravel 的 ->where(function($query) { ... }) 语法：

var query = User.query(&db);
defer query.deinit();

// Laravel: ->where('active', 1)->where(function($q) { 
//              $q->where('role', 'admin')->orWhere('role', 'mod') 
//          })
// SQL: WHERE active = 1 AND (role = 'admin' OR role = 'moderator')

_ = query.whereEq("active", 1).whereGroup(struct {
    fn apply(q: *@TypeOf(query)) void {
        _ = q.whereEq("role", "admin").orWhereEq("role", "moderator");
    }
}.apply);

嵌套条件构建器（更灵活）

var query = User.query(&db);
defer query.deinit();

// 创建嵌套条件
var nested = query.newNested();
_ = nested.whereEq("city", "北京").orWhereEq("city", "上海");

// 添加到主查询：WHERE active = 1 AND (city = '北京' OR city = '上海')
_ = query.whereEq("active", 1).whereNested(&nested);

// OR 嵌套：WHERE active = 1 OR (city = '北京' AND city = '上海')
// _ = query.whereEq("active", 1).orWhereNested(&nested);

完整插件示例

const std = @import("std");
const plugin = @import("plugin_interface.zig");

// 插件元信息
pub const info = plugin.PluginInfo{
    .name = "MyPlugin",
    .version = "1.0.0",
    .description = "我的第一个插件",
    .author = "开发者",
    .license = "MIT",
};

// 能力声明
pub const capabilities = plugin.PluginCapabilities{
    .http_handlers = true,
    .event_listener = true,
};

// 插件数据
const PluginData = struct {
    allocator: std.mem.Allocator,
    counter: u32 = 0,

    fn init(alloc: std.mem.Allocator) !*PluginData {
        const self = try alloc.create(PluginData);
        self.* = .{ .allocator = alloc };
        return self;
    }

    fn deinit(self: *PluginData) void {
        self.allocator.destroy(self);
    }
};

var g_data: ?*PluginData = null;

// 导出函数
export fn plugin_get_info() callconv(.C) *const plugin.PluginInfo {
    return &info;
}

export fn plugin_get_capabilities() callconv(.C) u32 {
    return capabilities.toBitmap();
}

export fn plugin_init(allocator: std.mem.Allocator) callconv(.C) ?*anyopaque {
    g_data = PluginData.init(allocator) catch return null;
    return @ptrCast(g_data);
}

export fn plugin_deinit(handle: ?*anyopaque) callconv(.C) void {
    _ = handle;
    if (g_data) |data| {
        data.deinit();
        g_data = null;
    }
}

export fn plugin_start(handle: ?*anyopaque) callconv(.C) bool {
    _ = handle;
    std.log.info("插件已启动", .{});
    return true;
}

export fn plugin_stop(handle: ?*anyopaque) callconv(.C) bool {
    _ = handle;
    std.log.info("插件已停止", .{});
    return true;
}
